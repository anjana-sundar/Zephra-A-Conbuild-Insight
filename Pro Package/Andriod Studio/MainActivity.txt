                        socket_vid = new Socket(IP_Text.getText().toString(), 6666);

                        while(!killStreaming){

                            while (total_buffer_len_till_now < yuv_buff_len ){
                                buffer_len = input_vid.read(temp_img_yuv_buff,0, (yuv_buff_len - total_buffer_len_till_now));
                                for (int count=0; count<buffer_len; count++){
                                    img_yuv_buff[total_buffer_len_till_now + count] = temp_img_yuv_buff[count];
                                }
                                total_buffer_len_till_now = total_buffer_len_till_now + buffer_len;
                            }
                            total_buffer_len_till_now = 0;



                           int uv_counter=0;
                            for(int cntH=0; cntH<UV_height; cntH++){
                                for(int cntW=0; cntW<UV_width; cntW++){
                                    y1 = (int)img_yuv_buff[cntH*width*2 + cntW*2];
                                    y2 = (int)img_yuv_buff[cntH*width*2 + cntW*2 + 1];
                                    y3 = (int)img_yuv_buff[cntH*width*2 + width + cntW*2];
                                    y4 = (int)img_yuv_buff[cntH*width*2 + width + cntW*2 + 1];

                                    u = (int)img_yuv_buff[Y_len + uv_counter];
                                    v = (int)img_yuv_buff[Y_len + U_or_V_len + uv_counter];

                                    R =  (int)(y1  + 1.402*( v - 128 ));
                                    G =  (int)(y1  - 0.344*( u - 128 ) - 0.714*( v - 128 ));
                                    B =  (int)(y1  + 1.772*( u - 128 ));
                                    if (R>255){R=225;} else if (R<0){R=0;}
                                    if (G>255){G=225;} else if (G<0){G=0;}
                                    if (B>255){B=225;} else if (B<0){B=0;}
                                    pixels[cntH*width*2 + cntW*2] = Color.rgb(R, G, B);
                                    R =  (int)(y2  + 1.402*( v - 128 ));
                                    G =  (int)(y2  - 0.344*( u - 128 ) - 0.714*( v - 128 ));
                                    B =  (int)(y2  + 1.772*( u - 128 ));
                                    if (R>255){R=225;} else if (R<0){R=0;}
                                    if (G>255){G=225;} else if (G<0){G=0;}
                                    if (B>255){B=225;} else if (B<0){B=0;}
                                    pixels[cntH*width*2 + cntW*2+1] = Color.rgb(R, G, B);
                                    R =  (int)(y3  + 1.402*( v - 128 ));
                                    G =  (int)(y3  - 0.344*( u - 128 ) - 0.714*( v - 128 ));

//                        pid_TV.setText("P:    " + df2.format((float)P_gain) + "       I:    " + df3.format((float)I_gain/100.0f) + "       D:    " + df3.format((float)D_gain/5.0f)

            Runnable get_ctrl_runnable = new Runnable() {
                @Override
                public void run() {
                    PrintWriter output;
                    BufferedReader input;
                    Socket socket;
                    try {
                        socket = new Socket(IP_Text.getText().toString(), 9191);
                        output = new PrintWriter(socket.getOutputStream());
                        input = new BufferedReader(new InputStreamReader(socket.getInputStream(),"ISO-8859-1"));
                        char send[] = new char[TCP_OUT_BUFF_SIZE];
                        char recv[] = new char[TCP_IN_BUFF_SIZE];
                        int confirm_Loop_Exit = 0;

                        while(true){

                            P_gain = P_seek.getProgress(); I_gain = I_seek.getProgress(); D_gain = D_seek.getProgress();
                            roll_offset = (int)( Roll_off_seek.getProgress() / 2 );
                            pitch_offset = (int)( Pitch_off_seek.getProgress() / 2 );

                            //----------------WRITE_DATA_UPDATE----------------//
                            send[0]= '$'; send[1]= (char)RJoy_X; send[2]= (char)RJoy_Y; send[3]= (char)throttle; send[4]= (char)Joy_yaw;
                            send[5] = Vehicle_cmd[0]; send[6] = Vehicle_cmd[1];
                            send[7]= (char)P_gain; send[8]= (char)I_gain; send[9]= (char)D_gain;
                            send[10]= (char)roll_offset; send[11]= (char)pitch_offset;
                            send[12]= '*';
                            //------------------------------------------------//


                            //-------------------READ_WRITE-------------------//
                            output.write(send, 0 ,TCP_OUT_BUFF_SIZE);
                            output.flush();
                            input.read(recv,0,TCP_IN_BUFF_SIZE);
                            String recv_string = String.valueOf(recv);
                            //------------------------------------------------//



                            //------------------RECV_DECODE-------------------//
                            if(recv[0] == '$' && recv[TCP_IN_BUFF_SIZE-1] == '*') {
                                //-----------------Quaternion_Angle_Calculation-----------------//
                                qw = (float) (((int) recv[1] / (float) 100) - 1);
                                qx = (float) (((int) recv[2] / (float) 100) - 1);
                                qy = (float) (((int) recv[3] / (float) 100) - 1);
                                qz = (float) (((int) recv[4] / (float) 100) - 1);
                                roll = (int) (57.2958f * asin(2 * (qw * qy - qz * qx)));
                                pitch = (int) (-57.2958f * atan2(2 * (qw * qx + qy * qz), 1 - 2 * (qx * qx + qy * qy)));
                                yaw = (int) (-57.2958f * atan2((2 * (qw * qz + qx * qy)), (1 - 2 * (qy * qy + qz * qz))));
                                //--------------------------------------------------------------//

                                //----------Update_Vehicle_Map_Date------------//

                                if (recv[9] == '.' && recv[20] == '.') {
                                    float temp = 0;
                                    temp += (float) (10 * (recv[7] - 48));
                                    temp += (float) (recv[8] - 48);
                                    temp += (float) ((float) (recv[10] - 48) / (float) 10);
                                    temp += (float) ((float) (recv[11] - 48) / (float) 100);
                                    temp += (float) ((float) (recv[12] - 48) / (float) 1000);
                                    temp += (float) ((float) (recv[13] - 48) / (float) 10000);
                                    temp += (float) ((float) (recv[14] - 48) / (float) 100000);
                                    temp = temp / (float) 60;
                                    Vehicle_gps_lattitude = (float) (10 * (recv[5] - 48)) + (float) (recv[6] - 48) + temp;
                                    temp = 0;

                                    temp += (float) (10 * (recv[18] - 48));
                                    temp += (float) (recv[19] - 48);
                                    temp += (float) ((float) (recv[21] - 48) / (float) 10);
                                    temp += (float) ((float) (recv[22] - 48) / (float) 100);
                                    temp += (float) ((float) (recv[23] - 48) / (float) 1000);
                                    temp += (float) ((float) (recv[24] - 48) / (float) 10000);
                                    temp += (float) ((float) (recv[25] - 48) / (float) 100000);
                                    temp = temp / (float) 60;
                                    Vehicle_gps_longitude = (float) (100 * (recv[15] - 48)) + (float) (10 * (recv[16] - 48)) + (float) (recv[17] - 48) + temp;
                                    temp = 0;


                                    Vehicle_gps_fix_type = (int) (recv[26] - 48);
                                    Vehicle_gps_Sat_count = (int) recv[27];
                                    Vehicle_gps_HDOP = recv_string.substring(28, 30);

                                } else {
                                    Vehicle_gps_lattitude = 0;
                                    Vehicle_gps_longitude = 0;
                                    Vehicle_gps_fix_type = 0;
                                    Vehicle_gps_Sat_count = 0;
                                    Vehicle_gps_HDOP = "0";
                                }

                                //----------------------Vehicle_Status----------------------//
                                Vehicle_status[0] = recv[31]; Vehicle_status[1] = recv[32];

                                Vehicle_velocity_X = ( (float)(int)Vehicle_status[0] - 128 )/10.0f;
                                Vehicle_velocity_Y = ( (float)(int)Vehicle_status[1] - 128 )/10.0f;

                                //----------------------------------------------------------//

                                map_bundle.putFloat("key_vehicle_gps_lattitude", Vehicle_gps_lattitude);
                                map_bundle.putFloat("key_vehicle_gps_longitude", Vehicle_gps_longitude);
                                map_bundle.putInt("key_vehicle_gps_fix_type", Vehicle_gps_fix_type);

                                //---------------------------------------------//
                            }
                            //------------------------------------------------//


                            //---------------------DELAY----------------------//
                            synchronized (this) {
                                try {
                                    wait(10);
                                } catch (InterruptedException e) {
                                    e.printStackTrace();
                                }
                            }  // delay
                            //------------------------------------------------//


                            if (!tcpLoop) { confirm_Loop_Exit++; Vehicle_cmd[0] = '!'; Vehicle_cmd[1] = '1'; }
                            if (confirm_Loop_Exit == 2) { break; }

                            handler.post(UpdateUI);    //UpdateUI

                        }
                    } catch (IOException e) {
                        e.printStackTrace();
                    }
                }
            };
            Thread tcp_ctrl_thread = new Thread(get_ctrl_runnable);
            tcp_ctrl_thread.start();
        });

            tcpLoop = false;
        });
